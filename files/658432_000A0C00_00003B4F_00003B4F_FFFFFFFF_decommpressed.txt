&НаКлиенте
Функция ПолучитьДанныеМаршрутаСледования(ИдентификаторСтроки)
	
	МаршрутСледования = Объект.ТаблицаМаршрутСледования.НайтиСтроки(Новый Структура(""GUIDСтроки"", ИдентификаторСтроки));

		Результат.Добавить(СтруктураМаршрутаСледования);
		
	КонецЦикла;
	
	Возврат Результат;

	Иначе
		Количество_Возврат = СтрокаДокумента.ВСД.Количество;
		СтрокиУпаковок_Возврат = СтрокаДокумента.ВСД.СписокФизическихУпаковок;

	Для Каждого ТекСтрока Из Объект.ТаблицаДанных Цикл
		
		Если МассивЗапрещенныхСтатусов.Найти(ТекСтрока.СтатусЗаявки) <> Неопределено Тогда
			Продолжить;
		КонецЕсли;
		
		Для Каждого ТекЗначение Из ДанныеЗаполнения Цикл
			
			Если ТекЗначение.Ключ = ""ОформитьПолныйВозврат"" Тогда
				
				Если ТекЗначение.Значение Тогда
					
					ЗаполнитьДанныеВозврата(ТекСтрока.ПолучитьИдентификатор());
					ТекСтрока.Количество = 0;
					ИнтеграцияВетисAPIКлиентСервер.ОчиститьПодчиненнуюТабличнуюЧасть(Объект, ""СписокФизическихУпаковок"", ТекСтрока.GUIDСтроки);

				КонецЕсли;
				
			Иначе
				
				ТекСтрока[ТекЗначение.Ключ] = ТекЗначение.Значение;
				
			КонецЕсли;
			
		КонецЦикла;
		
		ТекСтрока.Транспортировка   = СформироватьПредставлениеТранспортировки(ТекСтрока);
		ТекСтрока.АктНесоответствия = СформироватьПредставлениеАктаНесоответствия(ТекСтрока);
		ТекСтрока.Контроль          = СформироватьПредставлениеРезультатовКонтроля(ТекСтрока);

			ПредставлениеАктаНесоответствия = ПредставлениеАктаНесоответствия + "" № "" + СтрокаТабличнойЧасти.НомерАктаНесоответствия;
		КонецЕсли;
		
		Если ЗначениеЗаполнено(СтрокаТабличнойЧасти.ДатаАктаНесоответствия) Тогда
			ПредставлениеАктаНесоответствия = ПредставлениеАктаНесоответствия + "" от "" + Формат(СтрокаТабличнойЧасти.ДатаАктаНесоответствия, ""ДФ=dd.MM.yyyy"");
